    设计模式

    说到设计模式，经常听到，但是很少总结。

    从应用场景出发，这样更容易理解些

    • 单例模式

    场景：节省内存、数据源公用、方法公用等。开始我new一个app class，在项目全局都会用到它，它储存了项目的状态，版本号，公共属性等。假如多个app，那获取到的属性 数据就不是一个地方和相同结果的了。

    单例的生成一般都会是在构造方法中或指定一个方法来获取，存储到一个变量上(js中使用闭包，将变量保存在内存中)，判断变量是否存在

    function StuClass(){
}
let geStuClass = (function() {
    let e = null;
    return function() {
      // 内容还能在严谨 优化 这里简单处理
        if (!e) {
            e = new StuClass();
        }
        return e;
    } 
})();

// 我们多次 geStuClass() ,得到的都是第一次geStuClass()时 生成的StuClass对象

    # 策略模式

    应用场景:表单验证,不同的验证的条件不同,个数不同.需要一个机制，将不同的验证方法封装开来，表单用到哪些，把哪些方法放上来。

    
let ruleFunction = {
  // 判断大
  max(){},
  // 判断小
  min(){}
  // 判断是否为空
  emity(){}
  // ....
}

let ruleGroup = {
  'a':['max',',min']
}

// 验证
function check($a,$b){
  let _rules = ruleGroup[$a];
  for(let i in _rules){
    let ruleFunName =  _rules[i];
    if(!ruleFunction[ruleFunName]){
      return false;
    }
  }
  return true;
}

check('a',"123");

    # 迭代模式

    应用场景:按照指定顺序执行

    上方的策略模式 验证rule a 的下条件方法，则属于迭代模式，因为他是按照指定顺序执行的。其他的还有很多。

    # 发布订阅模式(观察者模式)

    应用场景:打开指定弹窗，我有很多弹窗组件，都有着指定的name。在弹窗组件生成时也订阅了方法，我通过打开方法，传递指定name，打开指定name的弹窗组件。

    
var events = {}
/**
 * 注册事件
 * @param {string} eventName 事件名称
 * @param {funciton} callback 回调方法
 */
export function register(eventName, callback) {
	if (typeof eventName !== "string") {
		return;
	}
	if (typeof callback !== "function") {
		return;
	}
	if (!events[eventName]) {
		events[eventName] = [];
	}
	events[eventName].push(callback);
}
/**
 * 触发事件
 * @param {string} eventName 事件名称
 * @param {object} data 触发参数
 */
export function trigger(eventName, data) {
	if (typeof eventName !== "string") {
		return;
	}
	if (!events[eventName]) {
		return;
	}
	for (var i in events[eventName]) {
		events[eventName][i](data);
	}
}

// 组件生成时
register('modal_name_001',()=>{
  this.show = true;
})
// 触发001弹窗
trigger('modal_name_001');
